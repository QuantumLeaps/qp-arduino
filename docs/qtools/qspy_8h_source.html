180  int n_ = SNPRINTF_S(&QSPY_output.buf[QS_LINE_OFFSET], \
181  (QS_LINE_LEN_MAX - QS_LINE_OFFSET), \
182  format_, ##__VA_ARGS__); \
183  if ((0 < n_) && (n_ < QS_LINE_LEN_MAX - QS_LINE_OFFSET)) { \
184  QSPY_output.len = n_; \
